return {
  {
    'echasnovski/mini.nvim',
    version = false,
    config = function()
      local copy_keymap = function(mode, from_lhs, to_lhs)
        local keymap = vim.fn.maparg(from_lhs, mode, false, true)
        if keymap and (keymap.callback or keymap.rhs) then
          vim.keymap.set(mode, to_lhs, keymap.callback or keymap.rhs, { desc = keymap.desc })
        end
      end
      copy_keymap('x', 'an', '<C-Up>')
      copy_keymap('x', 'in', '<C-Down>')

      require('mini.files').setup({
        mappings = {
          go_in = '<Right>',
          go_out = '<Left>',
          go_in_plus = '<S-Right>',
          go_out_plus = '<S-Left>'
        }
      })

      require("mini.comment").setup()
      require("mini.surround").setup()
      require("mini.pairs").setup()

      local ai = require("mini.ai")
      ai.setup({
        n_lines = 500,
        custom_textobjects = {
          ["="] = ai.gen_spec.treesitter({ a = "@assignment.outer", i = "@assignment.inner" }),
          ["L"] = ai.gen_spec.treesitter({ a = "@assignment.lhs",   i = "@assignment.lhs" }),
          ["R"] = ai.gen_spec.treesitter({ a = "@assignment.rhs",   i = "@assignment.rhs" }),
          ["a"] = ai.gen_spec.treesitter({ a = "@parameter.outer",  i = "@parameter.inner" }),
          ["i"] = ai.gen_spec.treesitter({ a = "@conditional.outer", i = "@conditional.inner" }),
          ["l"] = ai.gen_spec.treesitter({ a = "@loop.outer",        i = "@loop.inner" }),
          ["f"] = ai.gen_spec.treesitter({ a = "@call.outer",        i = "@call.inner" }),
          ["m"] = ai.gen_spec.treesitter({ a = "@function.outer",    i = "@function.inner" }),
          ["c"] = ai.gen_spec.treesitter({ a = "@class.outer",       i = "@class.inner" }),
        }
      })

      vim.api.nvim_create_autocmd('FileType', {
        callback = function()
          local lang = vim.treesitter.language.get_lang(vim.bo.filetype) or vim.bo.filetype
        end,
      })
    end,
  }
}
